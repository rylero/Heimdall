# Heimdall Milestone 4: ZeroMQ + Protobuf Communication Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bidirectional ZeroMQ push-pull communication between the Jetson and the robot (or mock robot), with Protobuf-encoded messages, heartbeat, and a `tools/mock_robot` program for local end-to-end testing without hardware.

**Architecture:** Two PUSH/PULL socket pairs — robot sends pose to Jetson (port 5555), Jetson sends track events to robot (port 5556). All messages serialized as Protobuf. `CommLayer` wraps both sockets with serialize/deserialize. `HeimdallApp` owns the full pipeline (DeepStream → PoseEstimator → ObjectTracker → CommLayer) and is the final integration target. Tasks 1–4 run on any machine; Task 5 requires Jetson.

**Tech Stack:** C++20, CMake FetchContent, libzmq v4.3.5, cppzmq v4.10.0, protobuf v26.1, Catch2 v3 (existing)

---

## ZeroMQ Topology

```
RoboRIO / mock_robot                    Jetson
──────────────────                   ──────────────
PUSH :5555  ──── robot pose ──────►  PULL *:5555
PULL :5556  ◄─── track events ────  PUSH *:5556
```

- Jetson **binds** both ports (persistent service).
- Robot / mock_robot **connects** to Jetson IP.
- Heartbeat: every outbound `DetectionFrameMsg` has `healthy=true`; pipeline error sets `healthy=false`.

## Protobuf Message Schema

```protobuf
// proto/heimdall.proto
syntax = "proto3";
package heimdall;

// RoboRIO → Jetson (every control loop, ~50 Hz)
message RobotPoseMsg {
    float  x           = 1;   // field-relative, meters
    float  y           = 2;
    float  heading     = 3;   // radians, CCW from +X
    uint64 timestamp_ns = 4;
}

// One tracked object in a frame
message TrackedObjectMsg {
    uint32 track_id   = 1;
    int32  class_id   = 2;
    float  x          = 3;   // field-relative, meters
    float  y          = 4;
    float  vx         = 5;   // velocity m/s
    float  vy         = 6;
    float  confidence = 7;
}

enum TrackEventTypeMsg {
    CONFIRMED = 0;
    UPDATED   = 1;
    LOST      = 2;
}

message TrackEventMsg {
    TrackEventTypeMsg type   = 1;
    TrackedObjectMsg  object = 2;
}

// Jetson → RoboRIO (every detection frame)
message DetectionFrameMsg {
    repeated TrackEventMsg events       = 1;
    uint64                 timestamp_ns = 2;
    bool                   healthy      = 3;   // heartbeat; false = pipeline error
}
```

```cpp
#pragma once
#include "pose/camera_params.h"
#include "tracker/track_event.h"
#include <optional>
#include <string>
#include <vector>
#include <zmq.hpp>

class CommLayer {
public:
    struct Config {
        std::string pose_bind_addr;    // Jetson PULL — receives robot pose (e.g. "tcp://*:5555")
        std::string output_bind_addr;  // Jetson PUSH — sends track events (e.g. "tcp://*:5556")
    };

    explicit CommLayer(Config config);

    // Non-blocking. Returns the next robot pose if one was waiting, nullopt otherwise.
    std::optional<RobotPose> try_recv_pose();

    // Serialize and send a frame of track events to the robot.
    // healthy=false signals a pipeline error (heartbeat failure).
    void send_frame(const std::vector<TrackEvent>& events,
                    uint64_t timestamp_ns,
                    bool healthy = true);

    // Expose context so tests can create peer sockets on the same context (inproc).
    zmq::context_t& context() { return ctx_; }

private:
    zmq::context_t ctx_;
    zmq::socket_t  pull_sock_;   // receives robot pose
    zmq::socket_t  push_sock_;   // sends track events
};
```

- [ ] **Step 4: Implement src/comm/comm_layer.cpp**

```cpp
#include "comm_layer.h"
#include "heimdall.pb.h"
#include <stdexcept>

CommLayer::CommLayer(Config config)
    : ctx_(1),
      pull_sock_(ctx_, zmq::socket_type::pull),
      push_sock_(ctx_, zmq::socket_type::push)
{
    // Receive timeout of 0 = non-blocking
    pull_sock_.set(zmq::sockopt::rcvtimeo, 0);
    pull_sock_.bind(config.pose_bind_addr);
    push_sock_.bind(config.output_bind_addr);
}

std::optional<RobotPose> CommLayer::try_recv_pose() {
    zmq::message_t msg;
    const auto result = pull_sock_.recv(msg, zmq::recv_flags::dontwait);
    if (!result) return std::nullopt;

    heimdall::RobotPoseMsg proto;
    if (!proto.ParseFromArray(msg.data(), static_cast<int>(msg.size())))
        return std::nullopt;

    return RobotPose{proto.x(), proto.y(), proto.heading()};
}

void CommLayer::send_frame(const std::vector<TrackEvent>& events,
                            uint64_t timestamp_ns,
                            bool healthy) {
    heimdall::DetectionFrameMsg frame;
    frame.set_timestamp_ns(timestamp_ns);
    frame.set_healthy(healthy);

    for (const auto& ev : events) {
        auto* msg_ev = frame.add_events();

        switch (ev.type) {
            case TrackEventType::CONFIRMED: msg_ev->set_type(heimdall::CONFIRMED); break;
            case TrackEventType::UPDATED:   msg_ev->set_type(heimdall::UPDATED);   break;
            case TrackEventType::LOST:      msg_ev->set_type(heimdall::LOST);      break;
        }

        auto* obj = msg_ev->mutable_object();
        obj->set_track_id(ev.object.track_id);
        obj->set_class_id(ev.object.class_id);
        obj->set_x(ev.object.x);
        obj->set_y(ev.object.y);
        obj->set_vx(ev.object.vx);
        obj->set_vy(ev.object.vy);
        obj->set_confidence(ev.object.confidence);
    }

    std::string bytes = frame.SerializeAsString();
    push_sock_.send(zmq::buffer(bytes), zmq::send_flags::none);
}
```

```cpp
#include <atomic>
#include <chrono>
#include <csignal>
#include <cstdio>
#include <cmath>
#include <string>
#include <thread>
#include <zmq.hpp>
#include "heimdall.pb.h"

static std::atomic<bool> g_running{true};

static void shutdown(int) { g_running = false; }

int main(int argc, char* argv[]) {
    // Optional: pass Jetson IP as argument (default = localhost)
    const std::string host = (argc > 1) ? argv[1] : "localhost";
    const std::string pose_addr   = "tcp://" + host + ":5555";
    const std::string output_addr = "tcp://" + host + ":5556";

    std::signal(SIGINT, shutdown);

    zmq::context_t ctx(1);

    // PUSH pose to Jetson PULL
    zmq::socket_t push_sock(ctx, zmq::socket_type::push);
    push_sock.connect(pose_addr);

    // PULL track events from Jetson PUSH
    zmq::socket_t pull_sock(ctx, zmq::socket_type::pull);
    pull_sock.set(zmq::sockopt::rcvtimeo, 1);  // 1 ms non-blocking
    pull_sock.connect(output_addr);

    std::printf("Mock robot connected to %s\n", host.c_str());
    std::printf("  Sending pose  → %s\n", pose_addr.c_str());
    std::printf("  Receiving from← %s\n\n", output_addr.c_str());

    uint64_t frame = 0;
    auto next_send = std::chrono::steady_clock::now();

    while (g_running) {
        auto now = std::chrono::steady_clock::now();
        if (now >= next_send) {
            // Simulate robot driving a slow circle
            const float t = static_cast<float>(frame) * 0.02f;  // seconds at 50 Hz
            heimdall::RobotPoseMsg pose;
            pose.set_x(3.f + 1.f * std::cos(t * 0.2f));
            pose.set_y(3.f + 1.f * std::sin(t * 0.2f));
            pose.set_heading(t * 0.2f);
            pose.set_timestamp_ns(static_cast<uint64_t>(t * 1e9));

            std::string bytes = pose.SerializeAsString();
            push_sock.send(zmq::buffer(bytes), zmq::send_flags::dontwait);

            ++frame;
            next_send += std::chrono::milliseconds(20);  // 50 Hz
        }

        // Non-blocking receive
        zmq::message_t msg;
        if (pull_sock.recv(msg, zmq::recv_flags::dontwait)) {
            heimdall::DetectionFrameMsg frame_msg;
            if (frame_msg.ParseFromArray(msg.data(), static_cast<int>(msg.size()))) {
                std::printf("[frame ts=%llu healthy=%d events=%d]\n",
                    static_cast<unsigned long long>(frame_msg.timestamp_ns()),
                    frame_msg.healthy(),
                    frame_msg.events_size());
                for (int i = 0; i < frame_msg.events_size(); ++i) {
                    const auto& ev  = frame_msg.events(i);
                    const auto& obj = ev.object();
                    const char* type_str =
                        ev.type() == heimdall::CONFIRMED ? "CONFIRMED" :
                        ev.type() == heimdall::UPDATED   ? "UPDATED"   : "LOST";
                    std::printf("  %s id=%u class=%d pos=(%.2f, %.2f) conf=%.2f\n",
                        type_str, obj.track_id(), obj.class_id(),
                        obj.x(), obj.y(), obj.confidence());
                }
            }
        }
    }

    std::printf("\nShutting down mock robot.\n");
    return 0;
}
```

- [ ] **Step 3: Implement src/app/heimdall_app.cpp**

```cpp
#include "heimdall_app.h"
#include <chrono>

HeimdallApp::HeimdallApp(Config config)
    : config_(std::move(config)),
      pose_estimator_(config_.cameras),
      tracker_(config_.tracker),
      comm_(config_.comm),
      pipeline_(config_.cameras, config_.infer_config_path,
                [this](const std::vector<Detection>& d){ on_detections(d); })
{}

HeimdallApp::~HeimdallApp() { stop(); }

void HeimdallApp::on_detections(const std::vector<Detection>& dets) {
    RobotPose pose;
    {
        std::lock_guard<std::mutex> lock(pose_mutex_);
        pose = latest_pose_;
    }

    auto field_dets = pose_estimator_.project(dets, pose);
    auto timestamp_ns = dets.empty() ? 0ULL : dets.front().timestamp_ns;

    auto events = tracker_.update(field_dets,
        static_cast<double>(timestamp_ns) * 1e-9);

    comm_.send_frame(events, timestamp_ns, /*healthy=*/true);
}

void HeimdallApp::pose_recv_loop() {
    while (running_) {
        if (auto pose = comm_.try_recv_pose()) {
            std::lock_guard<std::mutex> lock(pose_mutex_);
            latest_pose_ = *pose;
        }
        std::this_thread::sleep_for(std::chrono::microseconds(500));
    }
}

void HeimdallApp::run() {
    running_ = true;
    pose_recv_thread_ = std::thread([this]{ pose_recv_loop(); });
    pipeline_.run();  // blocks
}

void HeimdallApp::stop() {
    running_ = false;
    pipeline_.stop();
    if (pose_recv_thread_.joinable())
        pose_recv_thread_.join();
}
```

```cpp
static void shutdown(int) {

int main() {
    // ── Camera configuration ────────────────────────────────────────────────
    // Adjust for your physical camera mount.
    // rotation_from_euler(yaw, pitch, roll):
    //   yaw=0     → camera faces robot forward (+X)
    //   pitch=0.5 → camera tilted 0.5 rad (~28°) downward
    std::vector<CameraParams> cameras = {{
        .intrinsics = {500.f, 500.f, 320.f, 240.f},
        .extrinsics = {
            .tx = 0.3f, .ty = 0.f, .tz = 0.6f,  // 0.3m forward, 0.6m high
            .R  = rotation_from_euler(0.f, 0.5f, 0.f),
        },
    }};

        .cameras           = cameras,
        .infer_config_path = "config/infer_rfdetr.txt",
        .tracker           = {
            .confirmation_frames = 3,
            .loss_frames         = 5,
            .gate_distance       = 1.0f,
        },

            .pose_bind_addr   = "tcp://*:5555",
            .output_bind_addr = "tcp://*:5556",
        },
    };

    HeimdallApp app(cfg);
    g_app = &app;
    std::signal(SIGINT, shutdown);
```
